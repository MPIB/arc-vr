231  /// <summary>
232  /// Returns the status of a given BoolEvent
233  /// </summary>
234  public bool getEventStatus(BoolEvent type)
235  {
236  switch (type)
237  {
238  case BoolEvent.PRIMARY2DAXIS_CLICK : { return primary2DAxisClick; }
239  case BoolEvent.PRIMARY2DAXIS_CLICK_LEFT: { return primary2DAxisClick && primary2DAxis.x < -0.5; }
240  case BoolEvent.PRIMARY2DAXIS_CLICK_RIGHT: { return primary2DAxisClick && primary2DAxis.x > 0.5; }
241  case BoolEvent.PRIMARY2DAXIS_CLICK_LEFTORRIGHT: { return primary2DAxisClick && Mathf.Abs(primary2DAxis.x) > 0.5; }
242  case BoolEvent.PRIMARY2DAXIS_CLICK_MIDDLE: { return primary2DAxisClick && Mathf.Abs(primary2DAxis.x) < 0.5; }
243  case BoolEvent.PRIMARY2DAXIS_CLICKDOWN: { return primary2DAxisClickDown; }
244  case BoolEvent.PRIMARY2DAXIS_CLICKDOWN_LEFT: { return primary2DAxisClickDown && primary2DAxis.x < -0.5; }
245  case BoolEvent.PRIMARY2DAXIS_CLICKDOWN_RIGHT: { return primary2DAxisClickDown && primary2DAxis.x > 0.5; }
246  case BoolEvent.PRIMARY2DAXIS_CLICKDOWN_LEFTORRIGHT: { return primary2DAxisClickDown && Mathf.Abs(primary2DAxis.x) > 0.5; }
247  case BoolEvent.PRIMARY2DAXIS_CLICKDOWN_MIDDLE: { return primary2DAxisClickDown && Mathf.Abs(primary2DAxis.x) < 0.5; }
248  case BoolEvent.PRIMARY2DAXIS_TOUCH: { return primary2DAxisTouch; }
249  case BoolEvent.PRIMARY2DAXIS_TOUCH_LEFT: { return primary2DAxisTouch && primary2DAxis.x < -0.5; }
250  case BoolEvent.PRIMARY2DAXIS_TOUCH_RIGHT: { return primary2DAxisTouch && primary2DAxis.x > 0.5; }
251  case BoolEvent.PRIMARY2DAXIS_TOUCH_LEFTORRIGHT: { return primary2DAxisTouch && Mathf.Abs(primary2DAxis.x) > 0.5; }
252  case BoolEvent.PRIMARY2DAXIS_TOUCH_MIDDLE: { return primary2DAxisTouch && Mathf.Abs(primary2DAxis.x) < 0.5; }
253  case BoolEvent.TRIGGER_TRIGGER: { return triggerButton; }
254  case BoolEvent.TRIGGER_ONTRIGGERDOWN: { return triggerDown; }
255  case BoolEvent.TRIGGER_ONTRIGGERUP: { return triggerUp; }
256  case BoolEvent.MENUBUTTON: { return menuButton; }
257  case BoolEvent.MENUBUTTON_DOWN: { return menuButtonDown; }
258  case BoolEvent.MENUBUTTON_UP: { return menuButtonUp; }
259  case BoolEvent.GRIP_GRIP: { return grip; }
260  case BoolEvent.GRIP_ONGRIPDOWN: { return gripDown; }
261  case BoolEvent.GRIP_ONGRIPUP: { return gripUp; }
262  case BoolEvent.ANY_CANCEL: { return menuButtonDown || triggerDown || primary2DAxisClick; }
263  case BoolEvent.ALWAYS_TRUE: { return true; }
264  case BoolEvent.ALWAYS_FALSE: { return false; }
265  default : { AVR_DevConsole.cwarn("getEventStatus does not recoginze value "+type, this); break; }
266  }
267  return false;
268  }
