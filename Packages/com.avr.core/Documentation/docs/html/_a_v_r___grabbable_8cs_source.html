130  void UpdateVelocities()
131  {
132  #if AVR_NET
133  if(networkAPI.isOnline() && !networkAPI.isOwner(this)) return;
134  #endif
135 
136  // NOTE: use rb.worldCenterofMass etc isntead of transform pos + rotation ?
137  // TODO: Break grab if the object is too far away from hand
138 
139  // Get pos-rot of hand and item
140  Vector3 targetItemPosition = transform.position;
141  Quaternion targetItemRotation = transform.rotation;
142  Vector3 targetHandPosition = this.getTargetPosition();
143  Quaternion targetHandRotation = this.getTargetRotation();
144 
145  // Set the current objecttype to the regular object type or the nested one
146  _objType = (isGrabbedByMultipleHands && objectType.changeObjectTypeOnTwoHanded) ? objectType.typeOnTwoHanded : objectType;
147 
148  // Break joint if the distance is too far & it is growing.
149  if (Vector3.Distance(targetItemPosition, targetHandPosition) > last_dist && last_dist > _objType.Break_grab_distance)
150  {
151  while (AttachedHands.Count > 0) AttachedHands[0].makeRelease();
152  return;
153  }
154  last_dist = Vector3.Distance(targetItemPosition, targetHandPosition);
155 
156  // Follow aglorithms
157  switch(_objType.followType) {
158  case GrabbableObjectType.FollowType.FREE : {
159  //pos
160  Vector3 pDelta = (targetHandPosition - targetItemPosition);
161  Vector3 vel = pDelta / Time.fixedDeltaTime;
162 
163  rb.velocity = vel * _objType.Lightness;
164  lastVel = rb.velocity;
165 
166  //ang
167  Quaternion rotationDelta = targetHandRotation * Quaternion.Inverse(targetItemRotation);
168 
169  rotationDelta.ToAngleAxis(out float angle, out Vector3 axis);
170  while (angle > 180) angle -= 360; // Prevent object from doing sudden >180째 turns instead of negative <180째 ones
171 
172  rb.maxAngularVelocity = 99999.0f;
173 
174  Vector3 angvel = (angle * axis * Mathf.Deg2Rad) / Time.fixedDeltaTime;
175  if (!float.IsNaN(angvel.z)) rb.angularVelocity = angvel * _objType.Angular_Lightness;
176  break;
177  }
178  case GrabbableObjectType.FollowType.STATIC : {
179  break;
180  }
181  case GrabbableObjectType.FollowType.CONSTRAINED : {
182  Vector3 pDelta = (targetHandPosition - targetItemPosition);
183  Vector3 vel = pDelta / Time.fixedDeltaTime;
184 
185  // The current world acceleration is rb.velocity - lastVel. However, due to sudden changes in the objects position, we prevent the acceleration from jumping to
186  // dramatically by smoothing out the world acceleration throuhg a simple linear interpolation
187  // Lower the value of 0.5 to make the world acceleration more delayed/elastic, but also lower the jitter an object may experience
188  wacc = Vector3.Lerp(wacc, rb.velocity - lastVel, 0.5f);
189 
190  worldvel = wacc / Time.fixedDeltaTime; //We scale wacc similarly to pDelta, to compare with vel
191 
192  worldvel -= worldvel.normalized * Mathf.Min(worldvel.magnitude, 10); //10 == minimum force applied by hand
193  vel = Vector3.ClampMagnitude(vel, 30); //30 == maximum force applied by hand
194 
195  if (!float.IsNaN(worldvel.x) && !float.IsNaN(worldvel.y) && !float.IsNaN(worldvel.z))
196  {
197  rb.velocity = vel + worldvel;
198  }
199  else
200  {
201  rb.velocity = Vector3.zero;
202  }
203 
204  lastVel = rb.velocity;
205 
206 
207  //ang
208  Quaternion rotationDelta = targetHandRotation * Quaternion.Inverse(targetItemRotation);
209 
210  rotationDelta.ToAngleAxis(out float angle, out Vector3 axis);
211  while (angle > 180) angle -= 360; // Prevent object from doing sudden >180째 turns instead of negative <180째 ones
212 
213  rb.maxAngularVelocity = 99999.0f;
214 
215  Vector3 angvel = (angle * axis * Mathf.Deg2Rad) / Time.fixedDeltaTime;
216  if (!float.IsNaN(angvel.z)) rb.angularVelocity = (angvel * objectType.Angular_Lightness);
217  break;
218  }
219  case GrabbableObjectType.FollowType.HEAVY : {
220  // Point where object was grabbed
221  Vector3 closestp = AttachedHands[0].getWorldGrabLocation();
222 
223  // Difference between point where object was grabbed at and current hand/palm position (normalized over a few frames)
224  force = Vector3.Lerp(force, AttachedHands[0].grabPoint.position - closestp, 0.25f);
225 
226  // Velocity of rigidbody at grabbed position (normalized)
227  cvel = Vector3.Lerp(cvel, rb.GetPointVelocity(closestp), 0.25f);
228 
229  float k = 0.3f;
230  // delta = the force we want to apply minus the objects current velocity
231  Vector3 delta = force - Vector3.Lerp(Vector3.zero, cvel, k * Vector3.Distance(AttachedHands[0].grabPoint.position, closestp));
232  Vector3 f = delta * Time.fixedDeltaTime * 100.0f * _objType.Heavy_force_multiplier;
233 
234  rb.velocity *= 0.8f;
235  rb.AddForceAtPosition(f, closestp, ForceMode.Impulse);
236  break;
237  }
238  }
239  velocities.Add(rb.velocity);
240  while (velocities.Count > velocity_count) velocities.RemoveAt(0);
241  }
