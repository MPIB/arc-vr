56  public Vector3 naturalize_point(Vector3 wpos) {
57  try {
58  // Transform worldspace -> local space (relative to the PlayerRigs camera)
59  Vector3 cpos = AVR.Core.AVR_PlayerRig.Instance.MainCamera.transform.InverseTransformPoint(wpos);
60 
61  // Get a weighted average of reference points that are closest to this location
62  Vector3 closest = get_weighted_average_reference(cpos);
63 
64  // Smooth over time
65  outpos = Vector3.Lerp(outpos, closest, Time.deltaTime * smoothing_speed);
66 
67  // Interpolate by given value
68  return AVR.Core.AVR_PlayerRig.Instance.MainCamera.transform.TransformPoint(Vector3.Lerp(wpos, outpos, interp));
69  }
70  catch(System.Exception) {
71  AVR_DevConsole.cerror("PoseNaturalizationFilter failed! Is AVR_PlayerRig.Instance set correctly?", "AVR_PoseNaturalizationFilter");
72  return wpos;
73  }

76  Vector3 get_weighted_average_reference(Vector3 p)
77  {
78  // TODO: we are sorting the whole list here but only want the first k items. -> Needless performance cost.
79  List<Vector3> kn = _refpoints.OrderBy(item => Vector3.Distance(item, p)).Take(Mathf.Min(take_k, _refpoints.Count)).ToList();
80 
81  Vector3 sum = Vector3.zero;
82  float wsum = 0;
83  foreach (var v in kn)
84  {
85  float w = 1.0f / (0.0001f + Vector3.Distance(v, p));
86  wsum += w;
87  sum += w * v;
88  }
89 
90  sum /= wsum;
91 
92  return sum;
93  }
