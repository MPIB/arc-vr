232  Quaternion CorrectBodyYawAngle(Quaternion rot) {
233  float head_yaw = eyeTransform.localRotation.eulerAngles.y;
234  float body_yaw = last_yaw;
235 
236  // Get the difference in yaw between body and eyes/head
237  float yaw_diff = Mathf.DeltaAngle(head_yaw, body_yaw);
238 
239  // Account for max_head_yaw
240  yaw_diff = Mathf.Sign(yaw_diff) * Mathf.Max(0, Mathf.Abs(yaw_diff) - max_head_yaw);
241 
242  // Calcualte the speed at which we'll adapt.
243  float yaw_adapt_speed = Mathf.Abs(yaw_diff) * Time.deltaTime * 2.0f;
244 
245  // Adapt the rotation to the resulting yaw
246  return Quaternion.Euler(
247  bodyTransform.localRotation.eulerAngles.x,
248  Mathf.MoveTowardsAngle(body_yaw, head_yaw, yaw_adapt_speed),
249  bodyTransform.localRotation.eulerAngles.z
250  );
251  }
