63  location.grabbable.transform.rotation = grabbedObjectCenter.rotation;
64  grabbedObjectCenter.position = grabPoint.position + (location.grabbable.transform.position - location.location);

66  }

83  // This is a hacky/bad way of calculating the "entry-distance" (maximum distance inside the grabzone-collider in direction grabzone_p)

86  float entrydist = Vector3.Distance(smth, grabPoint.position);
