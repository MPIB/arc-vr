84  if (IsOwner)
85  {
86  bool changed = false;
87 
88  if (m_SyncPosition)

95  changed |= TryUpdate(netRotation, m_Rigidbody.rotation);
96  }

100  changed |= TryUpdate(netVelocity, m_Rigidbody.velocity);
101  }
102 
103  if (m_SyncAngularVelocity)
104  {
105  changed |= TryUpdate(netAngularVelocity, m_Rigidbody.angularVelocity);
106  }
107 
108  if (changed)
109  {
110  netUpdateId.Value++;
111  }
112  }
113  else
114  {
115  if (m_InterpolationChangeId != netUpdateId.Value)
116  {
117  BeginInterpolation();
118  m_InterpolationChangeId = netUpdateId.Value;
119  }
120 
121  float deltaTime = Time.fixedDeltaTime;
122  if (0 < m_InterpolationState.TimeRemaining)
123  {
124  deltaTime = Mathf.Min(deltaTime, m_InterpolationState.TimeRemaining);
125  m_InterpolationState.TimeRemaining -= deltaTime;
126 
127  deltaTime /= m_InterpolationState.TotalTime;
128 
129  if (m_SyncPosition)

131  m_Rigidbody.position +=
132  m_InterpolationState.PositionDelta * deltaTime;
133  }
134 
135  if (m_SyncRotation)
136  {
137  m_Rigidbody.rotation =
138  m_Rigidbody.rotation * Quaternion.Slerp(Quaternion.identity, m_InterpolationState.RotationDelta, deltaTime);
