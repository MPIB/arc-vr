46  public Quaternion get_target_rotation(Quaternion hand_rot) {
47  // Hand rotation * local node rotation (relative to the master grabbable). This is the rotation the grabbed object should have
48  // if allowed pitch, yaw and roll were to be 0
49  Quaternion zeroAngRot = hand_rot * (Quaternion.Inverse(transform.rotation) * master.transform.rotation);

55  Quaternion fromToRot = fullAngRot * Quaternion.Inverse(zeroAngRot);
